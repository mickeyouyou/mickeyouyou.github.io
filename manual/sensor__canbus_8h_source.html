126 template <typename SensorType>
127 Status SensorCanbus<SensorType>::Init() {
128  // load conf
129  if (!::apollo::common::util::GetProtoFromFile(FLAGS_sensor_conf_file,
130  &canbus_conf_)) {
131  return OnError("Unable to load canbus conf file: " +
132  FLAGS_sensor_conf_file);
133  }
134 
135  AINFO << "The canbus conf file is loaded: " << FLAGS_sensor_conf_file;
136  ADEBUG << "Canbus_conf:" << canbus_conf_.ShortDebugString();
137 
138  // Init can client
139  auto *can_factory = CanClientFactory::instance();
140  can_factory->RegisterCanClients();
141  can_client_ = can_factory->CreateCANClient(canbus_conf_.can_card_parameter());
142  if (!can_client_) {
143  return OnError("Failed to create can client.");
144  }
145  AINFO << "Can client is successfully created.";
146 
147  sensor_message_manager_ = std::unique_ptr<canbus::MessageManager<SensorType>>(
148  new canbus::MessageManager<SensorType>());
149  if (sensor_message_manager_ == nullptr) {
150  return OnError("Failed to create message manager.");
151  }
152  AINFO << "Sensor message manager is successfully created.";
153 
154  if (can_receiver_.Init(can_client_.get(), sensor_message_manager_.get(),
155  canbus_conf_.enable_receiver_log()) != ErrorCode::OK) {
156  return OnError("Failed to init can receiver.");
157  }
158  AINFO << "The can receiver is successfully initialized.";
159 
160  AdapterManager::Init(FLAGS_adapter_config_filename);
161 
162  AINFO << "The adapter manager is successfully initialized.";
163 
164  return Status::OK();
165 }

167 template <typename SensorType>
168 Status SensorCanbus<SensorType>::Start() {
169  // 1. init and start the can card hardware
170  if (can_client_->Start() != ErrorCode::OK) {
171  return OnError("Failed to start can client");
172  }
173  AINFO << "Can client is started.";
174 
175  // 2. start receive first then send
176  if (can_receiver_.Start() != ErrorCode::OK) {
177  return OnError("Failed to start can receiver.");
178  }
179  AINFO << "Can receiver is started.";
180 
181  // 3. set timer to triger publish info periodly
182  // if sensor_freq == 0, then it is event-triggered publishment.
183  // no need for timer.
184  if (FLAGS_sensor_freq > 0) {
185  const double duration = 1.0 / FLAGS_sensor_freq;
186  timer_ = AdapterManager::CreateTimer(
187  ros::Duration(duration), &SensorCanbus<SensorType>::OnTimer, this);
188  }
189 
190  // last step: publish monitor messages
191  apollo::common::monitor::MonitorBuffer buffer(&monitor_);
192  buffer.INFO("Canbus is started.");
193 
194  return Status::OK();
195 }
