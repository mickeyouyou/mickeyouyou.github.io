181 template <typename SensorType>
182 void MessageManager<SensorType>::Parse(const uint32_t message_id,
183  const uint8_t *data, int32_t length) {
184  ProtocolData<SensorType> *protocol_data =
185  GetMutableProtocolDataById(message_id);
186  if (protocol_data == nullptr) {
187  return;
188  }
189  {
190  std::lock_guard<std::mutex> lock(sensor_data_mutex_);
191  protocol_data->Parse(data, length, &sensor_data_);
192  }
193  received_ids_.insert(message_id);
194  // check if need to check period
195  const auto it = check_ids_.find(message_id);
196  if (it != check_ids_.end()) {
197  const int64_t time = apollo::common::time::AsInt64<micros>(Clock::Now());
198  it->second.real_period = time - it->second.last_time;
199  // if period 1.5 large than base period, inc error_count
200  const double period_multiplier = 1.5;
201  if (it->second.real_period > (it->second.period * period_multiplier)) {
202  it->second.error_count += 1;
203  } else {
204  it->second.error_count = 0;
205  }
206  it->second.last_time = time;
207  }
208 }
