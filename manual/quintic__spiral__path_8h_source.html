66  template <std::size_t N>
67  std::pair<double, double> DeriveCartesianDeviation(
68  const std::size_t param_index) const {
69  auto gauss_points = common::math::GetGaussLegendrePoints<N>();
70  std::array<double, N> x = gauss_points.first;
71  std::array<double, N> w = gauss_points.second;
72 
73  std::pair<double, double> cartesian_deviation = {0.0, 0.0};
74  for (std::size_t i = 0; i < N; ++i) {
75  double r = 0.5 * x[i] + 0.5;
76  double curr_theta = Evaluate(0, r * param_);
77  double derived_theta = DeriveTheta(param_index, r);
78 
79  cartesian_deviation.first +=
80  w[i] * (-std::sin(curr_theta)) * derived_theta;
81  cartesian_deviation.second += w[i] * std::cos(curr_theta) * derived_theta;
82  }
83 
84  cartesian_deviation.first *= param_ * 0.5;
85  cartesian_deviation.second *= param_ * 0.5;
86 
87  if (param_index == DELTA_S) {
88  for (std::size_t i = 0; i < N; ++i) {
89  double r = 0.5 * x[i] + 0.5;
90  double theta = Evaluate(0, r * param_);
91 
92  cartesian_deviation.first += 0.5 * w[i] * std::cos(theta);
93  cartesian_deviation.second += 0.5 * w[i] * std::sin(theta);
94  }
95  }
96  return cartesian_deviation;
97  }
