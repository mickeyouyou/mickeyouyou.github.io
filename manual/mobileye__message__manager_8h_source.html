63 template <>
64 ProtocolData<Mobileye>
65  *MessageManager<Mobileye>::GetMutableProtocolDataById(
66  const uint32_t message_id) {
67  uint32_t converted_message_id = message_id;
68  if (message_id >= 0x739 && message_id < (0x739 + 10 * 3)) {
69  converted_message_id = 0x739 + ((message_id - 0x739) % 3);
70  } else if (message_id >= 0x76c && message_id <= (0x76d + 7 * 2)) {
71  converted_message_id = 0x76c + ((message_id - 0x76c) % 2);
72  }
73 
74  if (protocol_data_map_.find(converted_message_id) ==
75  protocol_data_map_.end()) {
76  ADEBUG << "Unable to get protocol data because of invalid message_id:"
77  << message_id;
78  return nullptr;
79  }
80  return protocol_data_map_[converted_message_id];
81 }
