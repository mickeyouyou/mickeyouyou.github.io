87 ProtocolData<DelphiESR> *MessageManager<DelphiESR>::GetMutableProtocolDataById(
88  const uint32_t message_id) {
89  uint32_t converted_message_id = message_id;
90  if (message_id >= 0x500 && message_id <= 0x539) {
91  // repeated obstacle info
92  converted_message_id = 0x500;
93  }
94 
95  if (protocol_data_map_.find(converted_message_id) ==
96  protocol_data_map_.end()) {
97  ADEBUG << "Unable to get protocol data because of invalid message_id:"
98  << message_id;
99  return nullptr;
100  }
101  return protocol_data_map_[converted_message_id];
102 }

104 template <>
105 void MessageManager<DelphiESR>::Parse(const uint32_t message_id,
106  const uint8_t *data, int32_t length) {
107  ProtocolData<DelphiESR> *sensor_protocol_data =
108  GetMutableProtocolDataById(message_id);
109  if (sensor_protocol_data == nullptr) {
110  return;
111  }
112 
113  std::lock_guard<std::mutex> lock(sensor_data_mutex_);
114  sensor_protocol_data->Parse(data, length, &sensor_data_);
115 
116  // trigger publishment
117  if (message_id == 0x5E5) {
118  cvar_.notify_all();
119  }
120 
121  received_ids_.insert(message_id);
122  // check if need to check period
123  const auto it = check_ids_.find(message_id);
124  if (it != check_ids_.end()) {
125  const int64_t time = apollo::common::time::AsInt64<micros>(Clock::Now());
126  it->second.real_period = time - it->second.last_time;
127  // if period 1.5 large than base period, inc error_count
128  const double period_multiplier = 1.5;
129  if (it->second.real_period > (it->second.period * period_multiplier)) {
130  it->second.error_count += 1;
131  } else {
132  it->second.error_count = 0;
133  }
134  it->second.last_time = time;
135  }
136 }
