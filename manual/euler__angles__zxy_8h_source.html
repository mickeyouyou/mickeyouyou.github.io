168  T qw = cr * cp * cy - sr * sp * sy;
169  T qx = cr * sp * cy - sr * cp * sy;
170  T qy = cr * sp * sy + sr * cp * cy;
171  T qz = cr * cp * sy + sr * sp * cy;
172  if (qw < 0.0) return {-qw, -qx, -qy, -qz};
173  return {qw, qx, qy, qz};
