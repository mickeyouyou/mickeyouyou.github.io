52  bool Init(int rows, int cols, float range) {
53  rows_ = rows;
54  cols_ = cols;
55  grids_ = rows_ * cols_;
56  range_ = range;
57  scale_ = 0.5 * static_cast<float>(rows_) / range_;
58  inv_res_x_ = 0.5 * static_cast<float>(cols_) / range_;
59  inv_res_y_ = 0.5 * static_cast<float>(rows_) / range_;
60  point2grid_.clear();
61  obstacles_.clear();
62  id_img_.assign(grids_, -1);
63  pc_ptr_.reset();
64  valid_indices_in_pc_ = nullptr;
65  return true;

68  void Cluster(const caffe::Blob<float>& category_pt_blob,
69  const caffe::Blob<float>& instance_pt_blob,
70  const apollo::perception::pcl_util::PointCloudPtr& pc_ptr,
71  const apollo::perception::pcl_util::PointIndices& valid_indices,
72  float objectness_thresh,
73  bool use_all_grids_for_clustering) {
74  const float* category_pt_data = category_pt_blob.cpu_data();
75  const float* instance_pt_x_data = instance_pt_blob.cpu_data();
76  const float* instance_pt_y_data =
77  instance_pt_blob.cpu_data() + instance_pt_blob.offset(0, 1);
78 
79  pc_ptr_ = pc_ptr;
80  std::vector<std::vector<Node>> nodes(rows_,
81  std::vector<Node>(cols_, Node()));
82 
83  // map points into grids
84  size_t tot_point_num = pc_ptr_->size();
85  valid_indices_in_pc_ = &(valid_indices.indices);
86  CHECK_LE(valid_indices_in_pc_->size(), tot_point_num);
87  point2grid_.assign(valid_indices_in_pc_->size(), -1);
88 
89  for (size_t i = 0; i < valid_indices_in_pc_->size(); ++i) {
90  int point_id = valid_indices_in_pc_->at(i);
91  CHECK_GE(point_id, 0);
92  CHECK_LT(point_id, static_cast<int>(tot_point_num));
93  const auto& point = pc_ptr_->points[point_id];
94  // * the coordinates of x and y have been exchanged in feature generation
95  // step,
96  // so we swap them back here.
97  int pos_x = F2I(point.y, range_, inv_res_x_); // col
98  int pos_y = F2I(point.x, range_, inv_res_y_); // row
99  if (IsValidRowCol(pos_y, pos_x)) {
100  // get grid index and count point number for corresponding node
101  point2grid_[i] = RowCol2Grid(pos_y, pos_x);
102  nodes[pos_y][pos_x].point_num++;
103  }
104  }
105 
106  // construct graph with center offset prediction and objectness
107  for (int row = 0; row < rows_; row++) {
108  for (int col = 0; col < cols_; col++) {
109  int grid = RowCol2Grid(row, col);
110  Node* node = &nodes[row][col];
111  apollo::perception::DisjointSetMakeSet(node);
112  node->is_object =
113  (use_all_grids_for_clustering || nodes[row][col].point_num > 0) &&
114  (*(category_pt_data + grid) >= objectness_thresh);
115  int center_row = std::round(row + instance_pt_x_data[grid] * scale_);
116  int center_col = std::round(col + instance_pt_y_data[grid] * scale_);
117  center_row = std::min(std::max(center_row, 0), rows_ - 1);
118  center_col = std::min(std::max(center_col, 0), cols_ - 1);
119  node->center_node = &nodes[center_row][center_col];
120  }
121  }
122 
123  // traverse nodes
124  for (int row = 0; row < rows_; row++) {
125  for (int col = 0; col < cols_; col++) {
126  Node* node = &nodes[row][col];
127  if (node->is_object && node->traversed == 0) {
128  Traverse(node);
129  }
130  }
131  }
132  for (int row = 0; row < rows_; row++) {
133  for (int col = 0; col < cols_; col++) {
134  Node* node = &nodes[row][col];
135  if (!node->is_center) {
136  continue;
137  }
138  for (int row2 = row - 1; row2 <= row + 1; row2++) {
139  for (int col2 = col - 1; col2 <= col + 1; col2++) {
140  if ((row2 == row || col2 == col) && IsValidRowCol(row2, col2)) {
141  Node* node2 = &nodes[row2][col2];
142  if (node2->is_center) {
143  apollo::perception::DisjointSetUnion(node, node2);
144  }
145  }
146  }
147  }
148  }
149  }
150 
151  int count_obstacles = 0;
152  obstacles_.clear();
153  id_img_.assign(grids_, -1);
154  for (int row = 0; row < rows_; row++) {
155  for (int col = 0; col < cols_; col++) {
156  Node* node = &nodes[row][col];
157  if (!node->is_object) {
158  continue;
159  }
160  Node* root = apollo::perception::DisjointSetFind(node);
161  if (root->obstacle_id < 0) {
162  root->obstacle_id = count_obstacles++;
163  CHECK_EQ(static_cast<int>(obstacles_.size()), count_obstacles - 1);
164  obstacles_.push_back(Obstacle());
165  }
166  int grid = RowCol2Grid(row, col);
167  CHECK_GE(root->obstacle_id, 0);
168  id_img_[grid] = root->obstacle_id;
169  obstacles_[root->obstacle_id].grids.push_back(grid);
170  }
171  }
172  CHECK_EQ(static_cast<int>(count_obstacles), obstacles_.size());
173  }

175  void Filter(const caffe::Blob<float>& confidence_pt_blob,
176  const caffe::Blob<float>& height_pt_blob) {
177  const float* confidence_pt_data = confidence_pt_blob.cpu_data();
178  const float* height_pt_data = height_pt_blob.cpu_data();
179  for (size_t obstacle_id = 0; obstacle_id < obstacles_.size();
180  obstacle_id++) {
181  Obstacle* obs = &obstacles_[obstacle_id];
182  CHECK_GT(obs->grids.size(), 0);
183  double score = 0.0;
184  double height = 0.0;
185  for (int grid : obs->grids) {
186  score += static_cast<double>(confidence_pt_data[grid]);
187  height += static_cast<double>(height_pt_data[grid]);
188  }
189  obs->score = score / static_cast<double>(obs->grids.size());
190  obs->height = height / static_cast<double>(obs->grids.size());
191  obs->cloud.reset(new apollo::perception::pcl_util::PointCloud);
192  }
193  }

195  void GetObjects(const float confidence_thresh,
196  const float height_thresh,
197  const int min_pts_num,
198  std::vector<ObjectPtr>* objects) {
199  CHECK(valid_indices_in_pc_ != nullptr);
200 
201  for (size_t i = 0; i < point2grid_.size(); ++i) {
202  int grid = point2grid_[i];
203  if (grid < 0) {
204  continue;
205  }
206 
207  CHECK_GE(grid, 0);
208  CHECK_LT(grid, grids_);
209  int obstacle_id = id_img_[grid];
210 
211  int point_id = valid_indices_in_pc_->at(i);
212  CHECK_GE(point_id, 0);
213  CHECK_LT(point_id, static_cast<int>(pc_ptr_->size()));
214 
215  if (obstacle_id >= 0 &&
216  obstacles_[obstacle_id].score >= confidence_thresh) {
217  if (height_thresh < 0 ||
218  pc_ptr_->points[point_id].z <=
219  obstacles_[obstacle_id].height + height_thresh) {
220  obstacles_[obstacle_id].cloud->push_back(pc_ptr_->points[point_id]);
221  }
222  }
223  }
224 
225  for (size_t obstacle_id = 0; obstacle_id < obstacles_.size();
226  obstacle_id++) {
227  Obstacle* obs = &obstacles_[obstacle_id];
228  if (static_cast<int>(obs->cloud->size()) < min_pts_num) {
229  continue;
230  }
231  apollo::perception::ObjectPtr out_obj(new apollo::perception::Object);
232  out_obj->cloud = obs->cloud;
233  out_obj->score = obs->score;
234  objects->push_back(out_obj);
235  }
236  }

277  void Traverse(Node* x) {
278  std::vector<Node*> p;
279  p.clear();
280  while (x->traversed == 0) {
281  p.push_back(x);
282  x->traversed = 2;
283  x = x->center_node;
284  }
285  if (x->traversed == 2) {
286  for (int i = static_cast<int>(p.size()) - 1; i >= 0 && p[i] != x; i--) {
287  p[i]->is_center = true;
288  }
289  x->is_center = true;
290  }
291  for (size_t i = 0; i < p.size(); i++) {
292  Node* y = p[i];
293  y->traversed = 1;
294  y->parent = x->parent;
295  }
296  }
