31  static Eigen::Quaterniond RotateByMouse(T pre_x, T pre_y, T cur_x, T cur_y,
32  T obj_cen_x, T obj_cen_y,
33  T screen_width, T screen_height) {
34  double px = static_cast<double>(pre_x - obj_cen_x) / screen_width;
35  double py = static_cast<double>(obj_cen_y - pre_y) / screen_height;
36  double dx = static_cast<double>(cur_x - obj_cen_x) / screen_width;
37  double dy = static_cast<double>(obj_cen_y - cur_y) / screen_height;
38  const Eigen::Vector3d p1(px, py, ProjectToBall(px, py));
39  const Eigen::Vector3d p2(dx, dy, ProjectToBall(dx, dy));
40  Eigen::Vector3d axis = p2.cross(p1);
41  const double angle =
42  2.0 *
43  asin(sqrt(axis.squaredNorm() / p1.squaredNorm() / p2.squaredNorm()));
44  axis.normalize();
45  Eigen::AngleAxisd angleAxis(angle, axis);
46  return Eigen::Quaterniond(angleAxis);
47  }

49  static double ProjectToBall(double x, double y) {
50  const double rad = 1.0;
51  const double rad2 = rad * rad;
52  const double limit = rad2 * 0.5;
53 
54  const double d = x * x + y * y;
55  return d < limit ? sqrt(rad2 - d) : limit / sqrt(d);
56  }
