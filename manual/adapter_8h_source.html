365  // DumpMessage dumps the message to a file to
366  // /tmp/<adapter_name>/<name>.pb.txt, where the message is in ASCII
367  // mode and <name> is the .header().sequence_num() of the message.
368  template <typename InputMessageType>
369  bool DumpMessage(const enable_if_t<std::is_base_of<google::protobuf::Message,
370  InputMessageType>::value,
371  InputMessageType>& message) {
372  using google::protobuf::Message;
373  auto descriptor = message.GetDescriptor();
374  auto header_descriptor = descriptor->FindFieldByName("header");
375  if (header_descriptor == nullptr) {
376  ADEBUG << "Fail to find header field in pb.";
377  return false;
378  }
379  const Message& header = message.GetReflection()->GetMessage(
380  *static_cast<const Message*>(&message), header_descriptor);
381  auto seq_num_descriptor =
382  header.GetDescriptor()->FindFieldByName("sequence_num");
383  if (seq_num_descriptor == nullptr) {
384  ADEBUG << "Fail to find sequence_num field in pb.";
385  return false;
386  }
387  uint32_t sequence_num =
388  header.GetReflection()->GetUInt32(header, seq_num_descriptor);
389  return util::SetProtoToASCIIFile(
390  message, util::StrCat(dump_path_, "/", sequence_num, ".pb.txt"));
391  }

452  template <class T>
453  struct MessageDelay<
454  T, enable_if_t<std::is_base_of<google::protobuf::Message, T>::value>> {
455  static double Get(const T& new_msg, const T& last_msg) {
456  using google::protobuf::Message;
457 
458  return (ExtractTimeStampFromMsg(static_cast<const Message&>(new_msg)) -
459  ExtractTimeStampFromMsg(static_cast<const Message&>(last_msg))) *
460  1000.0;
461  }
462 
463  static double ExtractTimeStampFromMsg(
464  const google::protobuf::Message& message) {
465  using gpf = google::protobuf::FieldDescriptor;
466 
467  auto descriptor = message.GetDescriptor();
468  auto header_descriptor = descriptor->FindFieldByName("header");
469 
470  if (header_descriptor == nullptr ||
471  header_descriptor->cpp_type() != gpf::CPPTYPE_MESSAGE) {
472  return 0.0;
473  }
474 
475  auto timestamp_sec_descriptor =
476  header_descriptor->message_type()->FindFieldByName("timestamp_sec");
477  if (timestamp_sec_descriptor == nullptr ||
478  timestamp_sec_descriptor->cpp_type() != gpf::CPPTYPE_DOUBLE) {
479  return 0.0;
480  }
481 
482  const auto& header =
483  message.GetReflection()->GetMessage(message, header_descriptor);
484  return header.GetReflection()->GetDouble(header,
485  timestamp_sec_descriptor);
486  }
487  };
