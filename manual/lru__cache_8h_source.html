194  void Detach(Node<K, V>* node) {
195  if (node->prev != nullptr) {
196  node->prev->next = node->next;
197  }
198  if (node->next != nullptr) {
199  node->next->prev = node->prev;
200  }
201  node->prev = nullptr;
202  node->next = nullptr;
203  --size_;

206  void Attach(Node<K, V>* node) {
207  node->prev = &head_;

216  template <typename VV>
217  bool Update(const K& key, VV&& val, K* obs, bool add_only,
218  bool silent_update) {
219  if (obs == nullptr) {
220  return false;
221  }
222  if (Contains(key)) {
223  if (!add_only) {
224  map_[key].val = std::forward<VV>(val);
225  if (!silent_update) {
226  auto* node = &map_[key];
227  Detach(node);
228  Attach(node);
229  } else {
230  return false;
231  }
232  }
233  } else {
234  if (Full() && !GetObsolete(obs)) {
235  return false;
236  }
237 
238  map_.emplace(key, Node<K, V>(key, std::forward<VV>(val)));
239  Attach(&map_[key]);
240  }
241  return true;
242  }
243
