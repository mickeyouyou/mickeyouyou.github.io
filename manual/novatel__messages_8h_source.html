115 enum class SolutionStatus : uint32_t {

118  NO_CONVERGENCE, // no convergence
119  SINGULARITY, // singularity at parameters matrix

353 inline ImuParameter get_imu_parameter(ImuType type) {
354  switch (type) {
355  case ImuType::IMAR_FSAS:
356  // 0.1 * (2 ** -8) * (math.pi / 180 / 3600), (0.05 * (2 ** -15)
357  return {1.893803441835e-9, 1.52587890625e-6, 200.0};
358 
359  case ImuType::ADIS16488:
360  // 720/2**31 deg/LSB, 200/2**31 m/s/LSB
361  return {5.8516723170686385e-09, 9.31322574615478515625e-8, 200.0};
362 
363  case ImuType::STIM300:
364  // 2**-21 deg/LSB, 2**-22 m/s/LSB
365  return {8.32237840649762e-09, 2.384185791015625e-07, 125.0};
366 
367  case ImuType::ISA100:
368  case ImuType::ISA100C:
369  // 1.0e-9 rad/LSB, 2.0e-8 m/s/LSB
370  return {1.0e-9, 2.0e-8, 200.0};
371 
372  case ImuType::ISA100_400HZ:
373  case ImuType::ISA100C_400HZ:
374  return {1.0e-9, 2.0e-8, 400.0};
375 
376  default:
377  return {0.0, 0.0, 0.0};
378  }
379 }
