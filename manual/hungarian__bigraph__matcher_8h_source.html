46 class HungarianOptimizer {
47  static const int kHungarianOptimizerRowNotFound = -1;

50  public:
51  // Setup the initial conditions for the algorithm.

53  // Parameters: costs is a matrix of the cost of assigning each agent to
54  // each task. costs[i][j] is the cost of assigning agent i to task j.
55  // All the costs must be non-negative. This matrix does not have to
56  // be square (i.e. we can have different numbers of agents and tasks), but it
57  // must be regular (i.e. there must be the same number of entries in each row
58  // of the matrix).
59  explicit HungarianOptimizer(const std::vector<std::vector<double> >& costs);
60 
61  // Find an assignment which maximizes the total cost.
62  // Returns the assignment in the two vectors passed as argument.
63  // agent[i] is assigned to task[i].
64  void maximize(std::vector<int>* agent, std::vector<int>* task);
65 
66  // Find an assignment which minimizes the total cost.
67  // Returns the assignment in the two vectors passed as argument.
68  // agent[i] is assigned to task[i].
69  void minimize(std::vector<int>* agent, std::vector<int>* task);

71  private:
72  typedef void (HungarianOptimizer::*Step)();
73 
74  typedef enum { NONE, PRIME, STAR } Mark;
75 
76  // Convert the final cost matrix into a set of assignments of agents -> tasks.
77  // Returns the assignment in the two vectors passed as argument, the same as
78  // Minimize and Maximize
79  void find_assignments(std::vector<int>* agent, std::vector<int>* task);
80 
81  // Is the cell (row, col) starred?
82  bool is_starred(int row, int col) const { return marks_[row][col] == STAR; }
83 
84  // Mark cell (row, col) with a star
85  void star(int row, int col) {
86  marks_[row][col] = STAR;
87  stars_in_col_[col]++;
88  }
89 
90  // Remove a star from cell (row, col)
91  void unstar(int row, int col) {
92  marks_[row][col] = NONE;
93  stars_in_col_[col]--;
94  }
95

97  // kHungarianOptimizerColNotFound if no such column exists.
98  int find_star_in_row(int row) const;
99 
100  // Find a row in column 'col' containing a star, or return
101  // kHungarianOptimizerRowNotFound if no such row exists.
102  int find_star_in_col(int col) const;
103 
104  // Is cell (row, col) marked with a prime?
105  bool is_primed(int row, int col) const { return marks_[row][col] == PRIME; }
106 
107  // Mark cell (row, col) with a prime.
108  void prime(int row, int col) { marks_[row][col] = PRIME; }

151  // Run the Munkres algorithm!
152  void do_munkres();

154  void check_star();
155 
156  // Step 1.
157  // For each row of the matrix, find the smallest element and subtract it
158  // from every element in its row. Go to Step 2.
159  void reduce_rows();
160 
161  // Step 2.
162  // Find a zero (Z) in the matrix. If there is no starred zero in its row
163  // or column, star Z. Repeat for every element in the matrix. Go to step 3.
164  // Note: profiling shows this method to use 9.2% of the CPU - the next
165  // slowest step takes 0.6%. I can't think of a way of speeding it up though.
166  void star_zeroes();
167 
168  // Step 3.
169  // Cover each column containing a starred zero. If all columns are
170  // covered, the starred zeros describe a complete set of unique assignments.
171  // In this case, terminate the algorithm. Otherwise, go to step 4.
172  void cover_starred_zeroes();
173 
174  // Step 4.
175  // Find a noncovered zero and prime it. If there is no starred zero in the
176  // row containing this primed zero, Go to Step 5. Otherwise, cover this row
177  // and uncover the column containing the starred zero. Continue in this manner
178  // until there are no uncovered zeros left, then go to Step 6.
179  void prime_zeroes();
180 
181  // Step 5.
182  // Construct a series of alternating primed and starred zeros as follows.
183  // Let Z0 represent the uncovered primed zero found in Step 4. Let Z1 denote
184  // the starred zero in the column of Z0 (if any). Let Z2 denote the primed
185  // zero in the row of Z1 (there will always be one). Continue until the
186  // series terminates at a primed zero that has no starred zero in its column.
187  // Unstar each starred zero of the series, star each primed zero of the
188  // series, erase all primes and uncover every line in the matrix. Return to
189  // Step 3.
190  void make_augmenting_path();
191 
192  // Step 6.
193  // Add the smallest uncovered value in the matrix to every element of each
194  // covered row, and subtract it from every element of each uncovered column.
195  // Return to Step 4 without altering any stars, primes, or covered lines.
196  void augment_path();
