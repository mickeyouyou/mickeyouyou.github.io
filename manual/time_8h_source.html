241 // Measure run time of a code block, mostly for debugging puprpose.
242 // Example usage:
243 // PERF_BLOCK("Function Foo took: ") {
244 // Foo();
245 // }
246 // You can optionally pass in a time threshold (in second) so that the log will
247 // only be spit out when the elapsed time of running the code block is greater
248 // than it.
249 #define GET_MACRO(_1, _2, NAME, ...) NAME
250 #define PERF_BLOCK(...) \
251  GET_MACRO(__VA_ARGS__, PERF_BLOCK_WITH_THRESHOLD, PERF_BLOCK_NO_THRESHOLD) \
252  (__VA_ARGS__)
253 
254 #define PERF_BLOCK_NO_THRESHOLD(message) PERF_BLOCK_WITH_THRESHOLD(message, 0)
255 
256 #define PERF_BLOCK_WITH_THRESHOLD(message, threshold) \
257  using apollo::common::time::Clock; \
258  for (double block_start_time = 0; \
259  (block_start_time == 0 ? (block_start_time = Clock::NowInSecond()) \
260  : false); \
261  [&]() { \
262  double now = Clock::NowInSecond(); \
263  if (now - block_start_time > (threshold)) { \
264  AINFO << std::fixed << (message) << ": " << now - block_start_time \
265  << "s."; \
266  } \
267  }())
