46 /*
47  * @brief Returns heading (in radians) in [-PI, PI), with 0 being East.
48  * Note that x/y/z is East/North/Up.
49  *
50  * @param qw Quaternion w-coordinate
51  * @param qx Quaternion x-coordinate
52  * @param qy Quaternion y-coordinate
53  * @param qz Quaternion z-coordinate
54  *
55  * @return Heading encoded by given quaternion
56  */
57 inline double QuaternionToHeading(const double qw, const double qx,
58  const double qy, const double qz) {
59  EulerAnglesZXYd euler_angles(qw, qx, qy, qz);
60  // euler_angles.yaw() is zero when the car is pointing North, but
61  // the heading is zero when the car is pointing East.
62  return NormalizeAngle(euler_angles.yaw() + M_PI_2);
63 }

78 /*
79  * @brief Returns a quaternion encoding a rotation with zero roll, zero pitch,
80  * and the specified heading/yaw.
81  * Note that heading is zero at East and yaw is zero at North.
82  * Satisfies QuaternionToHeading(HeadingToQuaternion(h)) = h.
83  *
84  * @param heading The heading to encode in the rotation
85  *
86  * @return Quaternion encoding rotation by given heading
87  */
88 template <typename T>
89 inline Eigen::Quaternion<T> HeadingToQuaternion(const double heading) {
90  // Note that heading is zero at East and yaw is zero at North.
91  EulerAnglesZXY<T> euler_angles(heading - M_PI_2);
92  return euler_angles.ToQuaternion();
93 }

95 /*
96  * @brief Applies the rotation defined by a quaternion to a given vector.
97  * Note that x/y/z is East/North/Up.
98  *
99  * @param orientation Quaternion
100  * @param original Vector (in East-North-Up frame)
101  *
102  * @return Rotated vector
103  */
104 inline Eigen::Vector3d QuaternionRotate(const Quaternion &orientation,
105  const Eigen::Vector3d &original) {
106  Eigen::Quaternion<double> quaternion(orientation.qw(), orientation.qx(),
107  orientation.qy(), orientation.qz());
